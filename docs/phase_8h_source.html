74 {

78

81  // Phase array
82  doubleArray_t theta;
83  theta.resize(grid.m_xPoints, std::vector<double>(grid.m_yPoints));
84 
85  for (int num = 0; num < n_vort / 2; ++num)
86  {
87  doubleArray_t theta_k;
88  theta_k.resize(grid.m_xPoints, std::vector<double>(grid.m_yPoints));
89 
90  // Extract positions
91  auto[x_m, y_m] = positions[num];
92  auto[x_p, y_p] = positions[n_vort / 2 + num];
93 
94  double x_m_tilde = 2 * PI * ((x_m + grid.m_xLength) / grid.m_xLength);
95  double y_m_tilde = 2 * PI * ((y_m + grid.m_yLength) / grid.m_yLength);
96  double x_p_tilde = 2 * PI * ((x_p + grid.m_xLength) / grid.m_xLength);
97  double y_p_tilde = 2 * PI * ((y_p + grid.m_yLength) / grid.m_yLength);
98 
99 #pragma omp parallel for collapse(2) shared(grid, theta, theta_k, y_m_tilde, x_m_tilde, y_p_tilde, x_p_tilde) default(none)
100  for (int i = 0; i < grid.m_xPoints; ++i)
101  {
102  for (int j = 0; j < grid.m_yPoints; ++j)
103  {
104  double x_tilde = 2 * PI * ((grid.m_xMesh[i][j] + grid.m_xLength) / grid.m_xLength);
105  double y_tilde = 2 * PI * ((grid.m_yMesh[i][j] + grid.m_xLength) / grid.m_xLength);
106 
107  // Aux variables
108  double Y_minus = y_tilde - y_m_tilde;
109  double X_minus = x_tilde - x_m_tilde;
110  double Y_plus = y_tilde - y_p_tilde;
111  double X_plus = x_tilde - x_p_tilde;
112 
113  double heav_xp = heaviside(X_plus);
114  double heav_xm = heaviside(X_minus);
115 
116  for (int k = -5; k < 6; ++k)
117  {
118  theta_k[i][j] += std::atan(std::tanh((Y_minus + 2 * PI * k) / 2)
119  * std::tan((X_minus - PI) / 2))
120  - std::atan(std::tanh((Y_plus + 2 * PI * k) / 2)
121  * std::tan((X_plus - PI) / 2))
122  + PI * (heav_xp - heav_xm);
123  }
124 
125  theta_k[i][j] -= y_tilde * (x_p_tilde - x_m_tilde) / (2 * PI);
126  theta[i][j] += theta_k[i][j];
127  }
128  }
129 
130  }
